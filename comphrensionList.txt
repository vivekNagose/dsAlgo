def getTotalX(a, b):
    # Write your code here
    a.sort(); b.sort()
    af = [x for x in range(a[-1], b[0]+1) if all(x%i==0 for i in a)]
    bf = [x for x in range(a[-1], b[0]+1) if all(i%x==0 for i in b)]
    return len(set(af) & set(bf))
